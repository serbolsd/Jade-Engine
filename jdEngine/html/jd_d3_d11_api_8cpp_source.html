794  HRESULT
795  DirectX11Api::CreateInputLayoutDescFromVertexShaderSignature(ID3DBlob* pShaderBlob,
796  ID3D11Device* pD3DDevice,
797  ID3D11InputLayout** pInputLayout) {
798  // Reflect shader info
799  ID3D11ShaderReflection* pVertexShaderReflection = NULL;
800  if (FAILED(D3DReflect(pShaderBlob->GetBufferPointer(), pShaderBlob->GetBufferSize(), IID_ID3D11ShaderReflection, (void**)&pVertexShaderReflection)))
801  {
802  return S_FALSE;
803  }
804 
805  // Get shader info
806  D3D11_SHADER_DESC shaderDesc;
807  pVertexShaderReflection->GetDesc(&shaderDesc);
808 
809  // Read input layout description from shader info
810  std::vector<D3D11_INPUT_ELEMENT_DESC> inputLayoutDesc;
811  for (uint32 i = 0; i < shaderDesc.InputParameters; i++)
812  {
813  D3D11_SIGNATURE_PARAMETER_DESC paramDesc;
814  pVertexShaderReflection->GetInputParameterDesc(i, &paramDesc);
815 
816  // fill out input element desc
817  D3D11_INPUT_ELEMENT_DESC elementDesc;
818  elementDesc.SemanticName = paramDesc.SemanticName;
819  elementDesc.SemanticIndex = paramDesc.SemanticIndex;
820  elementDesc.InputSlot = 0;
821  elementDesc.AlignedByteOffset = D3D11_APPEND_ALIGNED_ELEMENT;
822  elementDesc.InputSlotClass = D3D11_INPUT_PER_VERTEX_DATA;
823  elementDesc.InstanceDataStepRate = 0;
824 
825  // determine DXGI format
826  if (paramDesc.Mask == 1)
827  {
828  if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_UINT32) elementDesc.Format = DXGI_FORMAT_R32_UINT;
829  else if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_SINT32) elementDesc.Format = DXGI_FORMAT_R32_SINT;
830  else if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_FLOAT32) elementDesc.Format = DXGI_FORMAT_R32_FLOAT;
831  }
832  else if (paramDesc.Mask <= 3)
833  {
834  if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_UINT32) elementDesc.Format = DXGI_FORMAT_R32G32_UINT;
835  else if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_SINT32) elementDesc.Format = DXGI_FORMAT_R32G32_SINT;
836  else if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_FLOAT32) elementDesc.Format = DXGI_FORMAT_R32G32_FLOAT;
837  }
838  else if (paramDesc.Mask <= 7)
839  {
840  if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_UINT32) elementDesc.Format = DXGI_FORMAT_R32G32B32_UINT;
841  else if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_SINT32) elementDesc.Format = DXGI_FORMAT_R32G32B32_SINT;
842  else if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_FLOAT32) elementDesc.Format = DXGI_FORMAT_R32G32B32_FLOAT;
843  }
844  else if (paramDesc.Mask <= 15)
845  {
846  if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_UINT32) elementDesc.Format = DXGI_FORMAT_R32G32B32A32_UINT;
847  else if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_SINT32) elementDesc.Format = DXGI_FORMAT_R32G32B32A32_SINT;
848  else if (paramDesc.ComponentType == D3D_REGISTER_COMPONENT_FLOAT32) elementDesc.Format = DXGI_FORMAT_R32G32B32A32_FLOAT;
849  }
850 
851  //save element desc
852  inputLayoutDesc.push_back(elementDesc);
853  }
854 
855  // Try to create Input Layout
856  HRESULT hr = pD3DDevice->CreateInputLayout(&inputLayoutDesc[0],
857  (UINT)inputLayoutDesc.size(),
858  pShaderBlob->GetBufferPointer(),
859  pShaderBlob->GetBufferSize(),
860  pInputLayout);
861 
862  //Free allocation shader reflection memory
863  pVertexShaderReflection->Release();
864  return hr;
865  }
