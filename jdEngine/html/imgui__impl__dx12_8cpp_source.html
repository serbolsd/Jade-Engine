71 static void ImGui_ImplDX12_SetupRenderState(ImDrawData* draw_data, ID3D12GraphicsCommandList* ctx, FrameResources* fr)
72 {
73  // Setup orthographic projection matrix into our constant buffer
74  // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right).
75  VERTEX_CONSTANT_BUFFER vertex_constant_buffer;
76  {
77  float L = draw_data->DisplayPos.x;
78  float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
79  float T = draw_data->DisplayPos.y;
80  float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
81  float mvp[4][4] =
82  {
83  { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
84  { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
85  { 0.0f, 0.0f, 0.5f, 0.0f },
86  { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
87  };
88  memcpy(&vertex_constant_buffer.mvp, mvp, sizeof(mvp));
89  }
90 
91  // Setup viewport
92  D3D12_VIEWPORT vp;
93  memset(&vp, 0, sizeof(D3D12_VIEWPORT));
94  vp.Width = draw_data->DisplaySize.x;
95  vp.Height = draw_data->DisplaySize.y;
96  vp.MinDepth = 0.0f;
97  vp.MaxDepth = 1.0f;
98  vp.TopLeftX = vp.TopLeftY = 0.0f;
99  ctx->RSSetViewports(1, &vp);
100 
101  // Bind shader and vertex buffers
102  unsigned int stride = sizeof(ImDrawVert);
103  unsigned int offset = 0;
104  D3D12_VERTEX_BUFFER_VIEW vbv;
105  memset(&vbv, 0, sizeof(D3D12_VERTEX_BUFFER_VIEW));
106  vbv.BufferLocation = fr->VertexBuffer->GetGPUVirtualAddress() + offset;
107  vbv.SizeInBytes = fr->VertexBufferSize * stride;
108  vbv.StrideInBytes = stride;
109  ctx->IASetVertexBuffers(0, 1, &vbv);
110  D3D12_INDEX_BUFFER_VIEW ibv;
111  memset(&ibv, 0, sizeof(D3D12_INDEX_BUFFER_VIEW));
112  ibv.BufferLocation = fr->IndexBuffer->GetGPUVirtualAddress();
113  ibv.SizeInBytes = fr->IndexBufferSize * sizeof(ImDrawIdx);
114  ibv.Format = sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT;
115  ctx->IASetIndexBuffer(&ibv);
116  ctx->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
117  ctx->SetPipelineState(g_pPipelineState);
118  ctx->SetGraphicsRootSignature(g_pRootSignature);
119  ctx->SetGraphicsRoot32BitConstants(0, 16, &vertex_constant_buffer, 0);
120 
121  // Setup blend factor
122  const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
123  ctx->OMSetBlendFactor(blend_factor);
124 }

127 // (this used to be set in io.RenderDrawListsFn and called by ImGui::Render(), but you can now call this directly from your main loop)
128 void ImGui_ImplDX12_RenderDrawData(ImDrawData* draw_data, ID3D12GraphicsCommandList* ctx)
129 {
130  // Avoid rendering when minimized
131  if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
132  return;
133 
134  // FIXME: I'm assuming that this only gets called once per frame!
135  // If not, we can't just re-allocate the IB or VB, we'll have to do a proper allocator.
136  g_frameIndex = g_frameIndex + 1;
137  FrameResources* fr = &g_pFrameResources[g_frameIndex % g_numFramesInFlight];
138 
139  // Create and grow vertex/index buffers if needed
140  if (fr->VertexBuffer == NULL || fr->VertexBufferSize < draw_data->TotalVtxCount)
141  {
142  SafeRelease(fr->VertexBuffer);
143  fr->VertexBufferSize = draw_data->TotalVtxCount + 5000;
144  D3D12_HEAP_PROPERTIES props;
145  memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
146  props.Type = D3D12_HEAP_TYPE_UPLOAD;
147  props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
148  props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
149  D3D12_RESOURCE_DESC desc;
150  memset(&desc, 0, sizeof(D3D12_RESOURCE_DESC));
151  desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
152  desc.Width = fr->VertexBufferSize * sizeof(ImDrawVert);
153  desc.Height = 1;
154  desc.DepthOrArraySize = 1;
155  desc.MipLevels = 1;
156  desc.Format = DXGI_FORMAT_UNKNOWN;
157  desc.SampleDesc.Count = 1;
158  desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
159  desc.Flags = D3D12_RESOURCE_FLAG_NONE;
160  if (g_pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc, D3D12_RESOURCE_STATE_GENERIC_READ, NULL, IID_PPV_ARGS(&fr->VertexBuffer)) < 0)
161  return;
162  }
163  if (fr->IndexBuffer == NULL || fr->IndexBufferSize < draw_data->TotalIdxCount)
164  {
165  SafeRelease(fr->IndexBuffer);
166  fr->IndexBufferSize = draw_data->TotalIdxCount + 10000;
167  D3D12_HEAP_PROPERTIES props;
168  memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
169  props.Type = D3D12_HEAP_TYPE_UPLOAD;
170  props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
171  props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
172  D3D12_RESOURCE_DESC desc;
173  memset(&desc, 0, sizeof(D3D12_RESOURCE_DESC));
174  desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
175  desc.Width = fr->IndexBufferSize * sizeof(ImDrawIdx);
176  desc.Height = 1;
177  desc.DepthOrArraySize = 1;
178  desc.MipLevels = 1;
179  desc.Format = DXGI_FORMAT_UNKNOWN;
180  desc.SampleDesc.Count = 1;
181  desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
182  desc.Flags = D3D12_RESOURCE_FLAG_NONE;
183  if (g_pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc, D3D12_RESOURCE_STATE_GENERIC_READ, NULL, IID_PPV_ARGS(&fr->IndexBuffer)) < 0)
184  return;
185  }
186 
187  // Upload vertex/index data into a single contiguous GPU buffer
188  void* vtx_resource, *idx_resource;
189  D3D12_RANGE range;
190  memset(&range, 0, sizeof(D3D12_RANGE));
191  if (fr->VertexBuffer->Map(0, &range, &vtx_resource) != S_OK)
192  return;
193  if (fr->IndexBuffer->Map(0, &range, &idx_resource) != S_OK)
194  return;
195  ImDrawVert* vtx_dst = (ImDrawVert*)vtx_resource;
196  ImDrawIdx* idx_dst = (ImDrawIdx*)idx_resource;
197  for (int n = 0; n < draw_data->CmdListsCount; n++)
198  {
199  const ImDrawList* cmd_list = draw_data->CmdLists[n];
200  memcpy(vtx_dst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
201  memcpy(idx_dst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
202  vtx_dst += cmd_list->VtxBuffer.Size;
203  idx_dst += cmd_list->IdxBuffer.Size;
204  }
205  fr->VertexBuffer->Unmap(0, &range);
206  fr->IndexBuffer->Unmap(0, &range);
207 
208  // Setup desired DX state
209  ImGui_ImplDX12_SetupRenderState(draw_data, ctx, fr);
210 
211  // Render command lists
212  // (Because we merged all buffers into a single one, we maintain our own offset into them)
213  int global_vtx_offset = 0;
214  int global_idx_offset = 0;
215  ImVec2 clip_off = draw_data->DisplayPos;
216  for (int n = 0; n < draw_data->CmdListsCount; n++)
217  {
218  const ImDrawList* cmd_list = draw_data->CmdLists[n];
219  for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
220  {
221  const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
222  if (pcmd->UserCallback != NULL)
223  {
224  // User callback, registered via ImDrawList::AddCallback()
225  // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
226  if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
227  ImGui_ImplDX12_SetupRenderState(draw_data, ctx, fr);
228  else
229  pcmd->UserCallback(cmd_list, pcmd);
230  }
231  else
232  {
233  // Apply Scissor, Bind texture, Draw
234  const D3D12_RECT r = { (LONG)(pcmd->ClipRect.x - clip_off.x), (LONG)(pcmd->ClipRect.y - clip_off.y), (LONG)(pcmd->ClipRect.z - clip_off.x), (LONG)(pcmd->ClipRect.w - clip_off.y) };
235  ctx->SetGraphicsRootDescriptorTable(1, *(D3D12_GPU_DESCRIPTOR_HANDLE*)&pcmd->TextureId);
236  ctx->RSSetScissorRects(1, &r);
237  ctx->DrawIndexedInstanced(pcmd->ElemCount, 1, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset, 0);
238  }
239  }
240  global_idx_offset += cmd_list->IdxBuffer.Size;
241  global_vtx_offset += cmd_list->VtxBuffer.Size;
242  }
243 }

245 static void ImGui_ImplDX12_CreateFontsTexture()
246 {
247  // Build texture atlas
248  ImGuiIO& io = ImGui::GetIO();
249  unsigned char* pixels;
250  int width, height;
251  io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
252 
253  // Upload texture to graphics system
254  {
255  D3D12_HEAP_PROPERTIES props;
256  memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
257  props.Type = D3D12_HEAP_TYPE_DEFAULT;
258  props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
259  props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
260 
261  D3D12_RESOURCE_DESC desc;
262  ZeroMemory(&desc, sizeof(desc));
263  desc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
264  desc.Alignment = 0;
265  desc.Width = width;
266  desc.Height = height;
267  desc.DepthOrArraySize = 1;
268  desc.MipLevels = 1;
269  desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
270  desc.SampleDesc.Count = 1;
271  desc.SampleDesc.Quality = 0;
272  desc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
273  desc.Flags = D3D12_RESOURCE_FLAG_NONE;
274 
275  ID3D12Resource* pTexture = NULL;
276  g_pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc,
277  D3D12_RESOURCE_STATE_COPY_DEST, NULL, IID_PPV_ARGS(&pTexture));
278 
279  UINT uploadPitch = (width * 4 + D3D12_TEXTURE_DATA_PITCH_ALIGNMENT - 1u) & ~(D3D12_TEXTURE_DATA_PITCH_ALIGNMENT - 1u);
280  UINT uploadSize = height * uploadPitch;
281  desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
282  desc.Alignment = 0;
283  desc.Width = uploadSize;
284  desc.Height = 1;
285  desc.DepthOrArraySize = 1;
286  desc.MipLevels = 1;
287  desc.Format = DXGI_FORMAT_UNKNOWN;
288  desc.SampleDesc.Count = 1;
289  desc.SampleDesc.Quality = 0;
290  desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
291  desc.Flags = D3D12_RESOURCE_FLAG_NONE;
292 
293  props.Type = D3D12_HEAP_TYPE_UPLOAD;
294  props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
295  props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
296 
297  ID3D12Resource* uploadBuffer = NULL;
298  HRESULT hr = g_pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc,
299  D3D12_RESOURCE_STATE_GENERIC_READ, NULL, IID_PPV_ARGS(&uploadBuffer));
300  IM_ASSERT(SUCCEEDED(hr));
301 
302  void* mapped = NULL;
303  D3D12_RANGE range = { 0, uploadSize };
304  hr = uploadBuffer->Map(0, &range, &mapped);
305  IM_ASSERT(SUCCEEDED(hr));
306  for (int y = 0; y < height; y++)
307  memcpy((void*) ((uintptr_t) mapped + y * uploadPitch), pixels + y * width * 4, width * 4);
308  uploadBuffer->Unmap(0, &range);
309 
310  D3D12_TEXTURE_COPY_LOCATION srcLocation = {};
311  srcLocation.pResource = uploadBuffer;
312  srcLocation.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
313  srcLocation.PlacedFootprint.Footprint.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
314  srcLocation.PlacedFootprint.Footprint.Width = width;
315  srcLocation.PlacedFootprint.Footprint.Height = height;
316  srcLocation.PlacedFootprint.Footprint.Depth = 1;
317  srcLocation.PlacedFootprint.Footprint.RowPitch = uploadPitch;
318 
319  D3D12_TEXTURE_COPY_LOCATION dstLocation = {};
320  dstLocation.pResource = pTexture;
321  dstLocation.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
322  dstLocation.SubresourceIndex = 0;
323 
324  D3D12_RESOURCE_BARRIER barrier = {};
325  barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
326  barrier.Flags = D3D12_RESOURCE_BARRIER_FLAG_NONE;
327  barrier.Transition.pResource = pTexture;
328  barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
329  barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_COPY_DEST;
330  barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
331 
332  ID3D12Fence* fence = NULL;
333  hr = g_pd3dDevice->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
334  IM_ASSERT(SUCCEEDED(hr));
335 
336  HANDLE event = CreateEvent(0, 0, 0, 0);
337  IM_ASSERT(event != NULL);
338 
339  D3D12_COMMAND_QUEUE_DESC queueDesc = {};
340  queueDesc.Type = D3D12_COMMAND_LIST_TYPE_DIRECT;
341  queueDesc.Flags = D3D12_COMMAND_QUEUE_FLAG_NONE;
342  queueDesc.NodeMask = 1;
343 
344  ID3D12CommandQueue* cmdQueue = NULL;
345  hr = g_pd3dDevice->CreateCommandQueue(&queueDesc, IID_PPV_ARGS(&cmdQueue));
346  IM_ASSERT(SUCCEEDED(hr));
347 
348  ID3D12CommandAllocator* cmdAlloc = NULL;
349  hr = g_pd3dDevice->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&cmdAlloc));
350  IM_ASSERT(SUCCEEDED(hr));
351 
352  ID3D12GraphicsCommandList* cmdList = NULL;
353  hr = g_pd3dDevice->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, cmdAlloc, NULL, IID_PPV_ARGS(&cmdList));
354  IM_ASSERT(SUCCEEDED(hr));
355 
356  cmdList->CopyTextureRegion(&dstLocation, 0, 0, 0, &srcLocation, NULL);
357  cmdList->ResourceBarrier(1, &barrier);
358 
359  hr = cmdList->Close();
360  IM_ASSERT(SUCCEEDED(hr));
361 
362  cmdQueue->ExecuteCommandLists(1, (ID3D12CommandList* const*)&cmdList);
363  hr = cmdQueue->Signal(fence, 1);
364  IM_ASSERT(SUCCEEDED(hr));
365 
366  fence->SetEventOnCompletion(1, event);
367  WaitForSingleObject(event, INFINITE);
368 
369  cmdList->Release();
370  cmdAlloc->Release();
371  cmdQueue->Release();
372  CloseHandle(event);
373  fence->Release();
374  uploadBuffer->Release();
375 
376  // Create texture view
377  D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc;
378  ZeroMemory(&srvDesc, sizeof(srvDesc));
379  srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
380  srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
381  srvDesc.Texture2D.MipLevels = desc.MipLevels;
382  srvDesc.Texture2D.MostDetailedMip = 0;
383  srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
384  g_pd3dDevice->CreateShaderResourceView(pTexture, &srvDesc, g_hFontSrvCpuDescHandle);
385  SafeRelease(g_pFontTextureResource);
386  g_pFontTextureResource = pTexture;
387  }
388 
389  // Store our identifier
390  static_assert(sizeof(ImTextureID) >= sizeof(g_hFontSrvGpuDescHandle.ptr), "Can't pack descriptor handle into TexID, 32-bit not supported yet.");
391  io.Fonts->TexID = (ImTextureID)g_hFontSrvGpuDescHandle.ptr;
392 }
