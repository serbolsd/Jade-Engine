60 static void ImGui_ImplDX10_SetupRenderState(ImDrawData* draw_data, ID3D10Device* ctx)
61 {
62  // Setup viewport
63  D3D10_VIEWPORT vp;
64  memset(&vp, 0, sizeof(D3D10_VIEWPORT));
65  vp.Width = (UINT)draw_data->DisplaySize.x;
66  vp.Height = (UINT)draw_data->DisplaySize.y;
67  vp.MinDepth = 0.0f;
68  vp.MaxDepth = 1.0f;
69  vp.TopLeftX = vp.TopLeftY = 0;
70  ctx->RSSetViewports(1, &vp);
71 
72  // Bind shader and vertex buffers
73  unsigned int stride = sizeof(ImDrawVert);
74  unsigned int offset = 0;
75  ctx->IASetInputLayout(g_pInputLayout);
76  ctx->IASetVertexBuffers(0, 1, &g_pVB, &stride, &offset);
77  ctx->IASetIndexBuffer(g_pIB, sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT, 0);
78  ctx->IASetPrimitiveTopology(D3D10_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
79  ctx->VSSetShader(g_pVertexShader);
80  ctx->VSSetConstantBuffers(0, 1, &g_pVertexConstantBuffer);
81  ctx->PSSetShader(g_pPixelShader);
82  ctx->PSSetSamplers(0, 1, &g_pFontSampler);
83  ctx->GSSetShader(NULL);
84 
85  // Setup render state
86  const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
87  ctx->OMSetBlendState(g_pBlendState, blend_factor, 0xffffffff);
88  ctx->OMSetDepthStencilState(g_pDepthStencilState, 0);
89  ctx->RSSetState(g_pRasterizerState);
90 }

93 // (this used to be set in io.RenderDrawListsFn and called by ImGui::Render(), but you can now call this directly from your main loop)
94 void ImGui_ImplDX10_RenderDrawData(ImDrawData* draw_data)
95 {
96  // Avoid rendering when minimized
97  if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
98  return;
99 
100  ID3D10Device* ctx = g_pd3dDevice;
101 
102  // Create and grow vertex/index buffers if needed
103  if (!g_pVB || g_VertexBufferSize < draw_data->TotalVtxCount)
104  {
105  if (g_pVB) { g_pVB->Release(); g_pVB = NULL; }
106  g_VertexBufferSize = draw_data->TotalVtxCount + 5000;
107  D3D10_BUFFER_DESC desc;
108  memset(&desc, 0, sizeof(D3D10_BUFFER_DESC));
109  desc.Usage = D3D10_USAGE_DYNAMIC;
110  desc.ByteWidth = g_VertexBufferSize * sizeof(ImDrawVert);
111  desc.BindFlags = D3D10_BIND_VERTEX_BUFFER;
112  desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
113  desc.MiscFlags = 0;
114  if (ctx->CreateBuffer(&desc, NULL, &g_pVB) < 0)
115  return;
116  }
117 
118  if (!g_pIB || g_IndexBufferSize < draw_data->TotalIdxCount)
119  {
120  if (g_pIB) { g_pIB->Release(); g_pIB = NULL; }
121  g_IndexBufferSize = draw_data->TotalIdxCount + 10000;
122  D3D10_BUFFER_DESC desc;
123  memset(&desc, 0, sizeof(D3D10_BUFFER_DESC));
124  desc.Usage = D3D10_USAGE_DYNAMIC;
125  desc.ByteWidth = g_IndexBufferSize * sizeof(ImDrawIdx);
126  desc.BindFlags = D3D10_BIND_INDEX_BUFFER;
127  desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
128  if (ctx->CreateBuffer(&desc, NULL, &g_pIB) < 0)
129  return;
130  }
131 
132  // Copy and convert all vertices into a single contiguous buffer
133  ImDrawVert* vtx_dst = NULL;
134  ImDrawIdx* idx_dst = NULL;
135  g_pVB->Map(D3D10_MAP_WRITE_DISCARD, 0, (void**)&vtx_dst);
136  g_pIB->Map(D3D10_MAP_WRITE_DISCARD, 0, (void**)&idx_dst);
137  for (int n = 0; n < draw_data->CmdListsCount; n++)
138  {
139  const ImDrawList* cmd_list = draw_data->CmdLists[n];
140  memcpy(vtx_dst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
141  memcpy(idx_dst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
142  vtx_dst += cmd_list->VtxBuffer.Size;
143  idx_dst += cmd_list->IdxBuffer.Size;
144  }
145  g_pVB->Unmap();
146  g_pIB->Unmap();
147 
148  // Setup orthographic projection matrix into our constant buffer
149  // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
150  {
151  void* mapped_resource;
152  if (g_pVertexConstantBuffer->Map(D3D10_MAP_WRITE_DISCARD, 0, &mapped_resource) != S_OK)
153  return;
154  VERTEX_CONSTANT_BUFFER* constant_buffer = (VERTEX_CONSTANT_BUFFER*)mapped_resource;
155  float L = draw_data->DisplayPos.x;
156  float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
157  float T = draw_data->DisplayPos.y;
158  float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
159  float mvp[4][4] =
160  {
161  { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
162  { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
163  { 0.0f, 0.0f, 0.5f, 0.0f },
164  { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
165  };
166  memcpy(&constant_buffer->mvp, mvp, sizeof(mvp));
167  g_pVertexConstantBuffer->Unmap();
168  }

170  // Backup DX state that will be modified to restore it afterwards (unfortunately this is very ugly looking and verbose. Close your eyes!)
171  struct BACKUP_DX10_STATE
172  {
173  UINT ScissorRectsCount, ViewportsCount;
174  D3D10_RECT ScissorRects[D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
175  D3D10_VIEWPORT Viewports[D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
176  ID3D10RasterizerState* RS;
177  ID3D10BlendState* BlendState;
178  FLOAT BlendFactor[4];
179  UINT SampleMask;
180  UINT StencilRef;
181  ID3D10DepthStencilState* DepthStencilState;
182  ID3D10ShaderResourceView* PSShaderResource;
183  ID3D10SamplerState* PSSampler;
184  ID3D10PixelShader* PS;
185  ID3D10VertexShader* VS;
186  ID3D10GeometryShader* GS;
187  D3D10_PRIMITIVE_TOPOLOGY PrimitiveTopology;
188  ID3D10Buffer* IndexBuffer, *VertexBuffer, *VSConstantBuffer;
189  UINT IndexBufferOffset, VertexBufferStride, VertexBufferOffset;
190  DXGI_FORMAT IndexBufferFormat;
191  ID3D10InputLayout* InputLayout;
192  };
193  BACKUP_DX10_STATE old;
194  old.ScissorRectsCount = old.ViewportsCount = D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE;
195  ctx->RSGetScissorRects(&old.ScissorRectsCount, old.ScissorRects);
196  ctx->RSGetViewports(&old.ViewportsCount, old.Viewports);
197  ctx->RSGetState(&old.RS);
198  ctx->OMGetBlendState(&old.BlendState, old.BlendFactor, &old.SampleMask);
199  ctx->OMGetDepthStencilState(&old.DepthStencilState, &old.StencilRef);
200  ctx->PSGetShaderResources(0, 1, &old.PSShaderResource);
201  ctx->PSGetSamplers(0, 1, &old.PSSampler);
202  ctx->PSGetShader(&old.PS);
203  ctx->VSGetShader(&old.VS);
204  ctx->VSGetConstantBuffers(0, 1, &old.VSConstantBuffer);
205  ctx->GSGetShader(&old.GS);
206  ctx->IAGetPrimitiveTopology(&old.PrimitiveTopology);
207  ctx->IAGetIndexBuffer(&old.IndexBuffer, &old.IndexBufferFormat, &old.IndexBufferOffset);
208  ctx->IAGetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset);
209  ctx->IAGetInputLayout(&old.InputLayout);
210 
211  // Setup desired DX state
212  ImGui_ImplDX10_SetupRenderState(draw_data, ctx);
213 
214  // Render command lists
215  // (Because we merged all buffers into a single one, we maintain our own offset into them)
216  int global_vtx_offset = 0;
217  int global_idx_offset = 0;
218  ImVec2 clip_off = draw_data->DisplayPos;
219  for (int n = 0; n < draw_data->CmdListsCount; n++)
220  {
221  const ImDrawList* cmd_list = draw_data->CmdLists[n];
222  for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
223  {
224  const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
225  if (pcmd->UserCallback)
226  {
227  // User callback, registered via ImDrawList::AddCallback()
228  // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
229  if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
230  ImGui_ImplDX10_SetupRenderState(draw_data, ctx);
231  else
232  pcmd->UserCallback(cmd_list, pcmd);
233  }
234  else
235  {
236  // Apply scissor/clipping rectangle
237  const D3D10_RECT r = { (LONG)(pcmd->ClipRect.x - clip_off.x), (LONG)(pcmd->ClipRect.y - clip_off.y), (LONG)(pcmd->ClipRect.z - clip_off.x), (LONG)(pcmd->ClipRect.w - clip_off.y)};
238  ctx->RSSetScissorRects(1, &r);
239 
240  // Bind texture, Draw
241  ID3D10ShaderResourceView* texture_srv = (ID3D10ShaderResourceView*)pcmd->TextureId;
242  ctx->PSSetShaderResources(0, 1, &texture_srv);
243  ctx->DrawIndexed(pcmd->ElemCount, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset);
244  }
245  }
246  global_idx_offset += cmd_list->IdxBuffer.Size;
247  global_vtx_offset += cmd_list->VtxBuffer.Size;
248  }
249 
250  // Restore modified DX state
251  ctx->RSSetScissorRects(old.ScissorRectsCount, old.ScissorRects);
252  ctx->RSSetViewports(old.ViewportsCount, old.Viewports);
253  ctx->RSSetState(old.RS); if (old.RS) old.RS->Release();
254  ctx->OMSetBlendState(old.BlendState, old.BlendFactor, old.SampleMask); if (old.BlendState) old.BlendState->Release();
255  ctx->OMSetDepthStencilState(old.DepthStencilState, old.StencilRef); if (old.DepthStencilState) old.DepthStencilState->Release();
256  ctx->PSSetShaderResources(0, 1, &old.PSShaderResource); if (old.PSShaderResource) old.PSShaderResource->Release();
257  ctx->PSSetSamplers(0, 1, &old.PSSampler); if (old.PSSampler) old.PSSampler->Release();
258  ctx->PSSetShader(old.PS); if (old.PS) old.PS->Release();
259  ctx->VSSetShader(old.VS); if (old.VS) old.VS->Release();
260  ctx->GSSetShader(old.GS); if (old.GS) old.GS->Release();
261  ctx->VSSetConstantBuffers(0, 1, &old.VSConstantBuffer); if (old.VSConstantBuffer) old.VSConstantBuffer->Release();
262  ctx->IASetPrimitiveTopology(old.PrimitiveTopology);
263  ctx->IASetIndexBuffer(old.IndexBuffer, old.IndexBufferFormat, old.IndexBufferOffset); if (old.IndexBuffer) old.IndexBuffer->Release();
264  ctx->IASetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset); if (old.VertexBuffer) old.VertexBuffer->Release();
265  ctx->IASetInputLayout(old.InputLayout); if (old.InputLayout) old.InputLayout->Release();
266 }

285  desc.SampleDesc.Count = 1;

287  desc.BindFlags = D3D10_BIND_SHADER_RESOURCE;

318  desc.AddressW = D3D10_TEXTURE_ADDRESS_WRAP;
319  desc.MipLODBias = 0.f;
320  desc.ComparisonFunc = D3D10_COMPARISON_ALWAYS;
321  desc.MinLOD = 0.f;
322  desc.MaxLOD = 0.f;
323  g_pd3dDevice->CreateSamplerState(&desc, &g_pFontSampler);
324  }
325 }

327 bool ImGui_ImplDX10_CreateDeviceObjects()
328 {
329  if (!g_pd3dDevice)
330  return false;
331  if (g_pFontSampler)
332  ImGui_ImplDX10_InvalidateDeviceObjects();
333 
334  // By using D3DCompile() from <d3dcompiler.h> / d3dcompiler.lib, we introduce a dependency to a given version of d3dcompiler_XX.dll (see D3DCOMPILER_DLL_A)
335  // If you would like to use this DX10 sample code but remove this dependency you can:
336  // 1) compile once, save the compiled shader blobs into a file or source code and pass them to CreateVertexShader()/CreatePixelShader() [preferred solution]
337  // 2) use code to detect any version of the DLL and grab a pointer to D3DCompile from the DLL.
338  // See https://github.com/ocornut/imgui/pull/638 for sources and details.
339 
340  // Create the vertex shader
341  {
342  static const char* vertexShader =
343  "cbuffer vertexBuffer : register(b0) \
344  {\
345  float4x4 ProjectionMatrix; \
346  };\
347  struct VS_INPUT\
348  {\
349  float2 pos : POSITION;\
350  float4 col : COLOR0;\
351  float2 uv : TEXCOORD0;\
352  };\
353  \
354  struct PS_INPUT\
355  {\
356  float4 pos : SV_POSITION;\
357  float4 col : COLOR0;\
358  float2 uv : TEXCOORD0;\
359  };\
360  \
361  PS_INPUT main(VS_INPUT input)\
362  {\
363  PS_INPUT output;\
364  output.pos = mul( ProjectionMatrix, float4(input.pos.xy, 0.f, 1.f));\
365  output.col = input.col;\
366  output.uv = input.uv;\
367  return output;\
368  }";
369 
370  ID3DBlob* vertexShaderBlob;
371  if (FAILED(D3DCompile(vertexShader, strlen(vertexShader), NULL, NULL, NULL, "main", "vs_4_0", 0, 0, &vertexShaderBlob, NULL)))
372  return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
373  if (g_pd3dDevice->CreateVertexShader(vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &g_pVertexShader) != S_OK)
374  {
375  vertexShaderBlob->Release();
376  return false;
377  }
378 
379  // Create the input layout
380  D3D10_INPUT_ELEMENT_DESC local_layout[] =
381  {
382  { "POSITION", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)IM_OFFSETOF(ImDrawVert, pos), D3D10_INPUT_PER_VERTEX_DATA, 0 },
383  { "TEXCOORD", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)IM_OFFSETOF(ImDrawVert, uv), D3D10_INPUT_PER_VERTEX_DATA, 0 },
384  { "COLOR", 0, DXGI_FORMAT_R8G8B8A8_UNORM, 0, (UINT)IM_OFFSETOF(ImDrawVert, col), D3D10_INPUT_PER_VERTEX_DATA, 0 },
385  };
386  if (g_pd3dDevice->CreateInputLayout(local_layout, 3, vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &g_pInputLayout) != S_OK)
387  {
388  vertexShaderBlob->Release();
389  return false;
390  }
391  vertexShaderBlob->Release();
392 
393  // Create the constant buffer
394  {
395  D3D10_BUFFER_DESC desc;
396  desc.ByteWidth = sizeof(VERTEX_CONSTANT_BUFFER);
397  desc.Usage = D3D10_USAGE_DYNAMIC;
398  desc.BindFlags = D3D10_BIND_CONSTANT_BUFFER;
399  desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
400  desc.MiscFlags = 0;
401  g_pd3dDevice->CreateBuffer(&desc, NULL, &g_pVertexConstantBuffer);
402  }
403  }
404 
405  // Create the pixel shader
406  {
407  static const char* pixelShader =
408  "struct PS_INPUT\
409  {\
410  float4 pos : SV_POSITION;\
411  float4 col : COLOR0;\
412  float2 uv : TEXCOORD0;\
413  };\
414  sampler sampler0;\
415  Texture2D texture0;\
416  \
417  float4 main(PS_INPUT input) : SV_Target\
418  {\
419  float4 out_col = input.col * texture0.Sample(sampler0, input.uv); \
420  return out_col; \
421  }";
422 
423  ID3DBlob* pixelShaderBlob;
424  if (FAILED(D3DCompile(pixelShader, strlen(pixelShader), NULL, NULL, NULL, "main", "ps_4_0", 0, 0, &pixelShaderBlob, NULL)))
425  return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
426  if (g_pd3dDevice->CreatePixelShader(pixelShaderBlob->GetBufferPointer(), pixelShaderBlob->GetBufferSize(), &g_pPixelShader) != S_OK)
427  {
428  pixelShaderBlob->Release();
429  return false;

432  }

434  // Create the blending setup
435  {
436  D3D10_BLEND_DESC desc;
437  ZeroMemory(&desc, sizeof(desc));
438  desc.AlphaToCoverageEnable = false;
439  desc.BlendEnable[0] = true;

441  desc.DestBlend = D3D10_BLEND_INV_SRC_ALPHA;

454  desc.FillMode = D3D10_FILL_SOLID;
455  desc.CullMode = D3D10_CULL_NONE;
456  desc.ScissorEnable = true;

461  // Create depth-stencil State
462  {
463  D3D10_DEPTH_STENCIL_DESC desc;
464  ZeroMemory(&desc, sizeof(desc));
465  desc.DepthEnable = false;
466  desc.DepthWriteMask = D3D10_DEPTH_WRITE_MASK_ALL;
467  desc.DepthFunc = D3D10_COMPARISON_ALWAYS;
